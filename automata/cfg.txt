/* MAIN STATE */
MAIN_STATE -> MAIN_STATE MAIN_STATE | EPSILON | IF_STATE | SWITCH_STATE | LOOP_STATE | FUNCTION_STATE | FUNCTION_CALL | TRY_STATE | THROW_STATE | DECLARATION_STATE | ASSIGNMENT

/* CONDITIONAL */
IF_STATE -> IF ROUND_BRACKET_OPEN EXPR ROUND_BRACKET_CLOSE CURLY_BRACKET_OPEN MAIN_STATE CURLY_BRACKET_CLOSE | IF ROUND_BRACKET_OPEN EXPR ROUND_BRACKET_CLOSE CURLY_BRACKET_OPEN MAIN_STATE CURLY_BRACKET_CLOSE ELSE_STATE 
ELSE_STATE -> ELSE IF_STATE | ELSE CURLY_BRACKET_OPEN MAIN_STATE CURLY_BRACKET_CLOSE
SWITCH_STATE -> SWITCH ROUND_BRACKET_OPEN VAR_NAME ROUND_BRACKET_CLOSE CURLY_BRACKET_OPEN CASE_STATE CURLY_BRACKET_CLOSE
CASE_STATE -> CASE_STATE CASE_STATE | CASE OPS_TYPE COLON MAIN_STATE BREAK | DEFAULT COLON MAIN_STATE BREAK | CASE OPS_TYPE COLON MAIN_STATE | DEFAULT COLON MAIN_STATE | EPSILON

/* LOOPING */
LOOP_STATE -> FOR_STATE | WHILE_STATE
FOR_STATE -> FOR ROUND_BRACKET_OPEN FOR_HEAD ROUND_BRACKET_CLOSE CURLY_BRACKET_OPEN LOOP_BODY CURLY_BRACKET_CLOSE
FOR_HEAD -> FOR_INIT EXPR EXPR | FOR_INIT EXPR ASSIGNMENT
FOR_INIT -> LET_DECLARATION | VAR_DECLARATION | ASSIGNMENT | EPSILON
WHILE_STATE -> WHILE ROUND_BRACKET_OPEN EXPR ROUND_BRACKET_CLOSE CURLY_BRACKET_OPEN LOOP_BODY CURLY_BRACKET_CLOSE
LOOP_BODY -> BREAK | CONTINUE | MAIN_STATE | EPSILON

/* FUNCTION */
FUNCTION_CALL -> VAR_NAME ROUND_BRACKET_OPEN PARAMETER ROUND_BRACKET_CLOSE
FUNCTION_STATE -> FUNCTION VAR_NAME ROUND_BRACKET_OPEN PARAMETER ROUND_BRACKET_CLOSE CURLY_BRACKET_OPEN FUNCTION_BODY CURLY_BRACKET_CLOSE
FUNCTION_BODY -> MAIN_STATE | RETURN OPS_TYPE | RETURN | EPSILON

/* TRY CATCH */
TRY_STATE -> TRY_BODY CATCH_BODY | TRY_BODY FINALLY_BODY | TRY_BODY CATCH_BODY FINALLY_BODY
TRY_BODY -> TRY CURLY_BRACKET_OPEN MAIN_STATE CURLY_BRACKET_CLOSE
CATCH_BODY -> CATCH ROUND_BRACKET_OPEN VAR_NAME ROUND_BRACKET_CLOSE CURLY_BRACKET_OPEN MAIN_STATE CURLY_BRACKET_CLOSE
FINALLY_BODY -> FINALLY CURLY_BRACKET_OPEN MAIN_STATE CURLY_BRACKET_CLOSE

/* THROW */
THROW_STATE -> THROW OPS_TYPE

/* PARAMETER */
PARAMETER -> PARAMETER COMMA PARAMETER | PARAM | EPSILON
PARAM -> VAR_NAME | ASSIGNMENT | EXPR | STRING | FUNCTION_CALL | BOOLEAN | NAN | NULL

/* VARIABLE */
ASSIGNMENT -> VAR_NAME ASSIGNMENT_SYMBOL OPS_TYPE | VAR_NAME ASSIGNMENT_SYMBOL CURLY_BRACKET_OPEN OBJECT_DECLARATION CURLY_BRACKET_CLOSE
ASSIGNMENT_SYMBOL -> EQUALS | PLUS EQUALS | MIN EQUALS | MULTIPLY EQUALS | DIVIDE EQUALS | MOD EQUALS | MULTIPLY MUTIPLY EQUALS | RIGHT RIGHT EQUALS | LEFT LEFT EQUALS | LEFT LEFT LEFT EQUALS | AND EQUALS | OR EQUALS | XOR EQUALS | AND AND EQUALS | OR OR EQUALS | QUESTION QUESTION EQUALS
DELETE_STATE -> DELETE VAR_NAME DOT VAR_NAME | DELETE VAR_NAME OPEN_SQUARE_BRACKET VAR_NAME CLOSE_SQUARE_BRACKET
DECLARATION_STATE -> CONST_DECLARATION | LET_DECLARATION | VAR_DECLARATION | ARRAY_DECLARATION | OBJECT_DECLARATION
CONST_DECLARATION -> CONST ASSIGNMENT
LET_DECLARATION -> LET ASSIGNMENT | LET VAR_NAME
VAR_DECLARATION -> VAR ASSIGNMENT | VAR VAR_NAME
ARRAY_DECLARATION -> VAR_NAME ASSIGNMENT_SYMBOL LIST_DECLARATION
OBJECT_DECLARATION -> OBJECT_DECLARATION COMMA OBJECT_DECLARATION | VAR_NAME COLON OPS_TYPE
LIST_DECLARATION -> OPEN_SQUARE_BRACKET PARAMETER CLOSE_SQUARE_BRACKET
LIST_ACCESS -> VAR_NAME OPEN_SQUARE_BRACKET EXPR CLOSE_SQUARE_BRACKET
OPS_TYPE -> PARAM | EXPR
BOOLEAN -> TRUE | FALSE

/* TERMINAL */
EPSILON -> epsilon
IF -> if
EXPR -> expr
CURLY_BRACKET_OPEN -> {
CURLY_BRACKET_CLOSE -> }
OPEN_SQUARE_BRACKET -> [
CLOSE_SQUARE_BRACKET -> ]
ROUND_BRACKET_OPEN -> (
ROUND_BRACKET_CLOSE -> )
ELSE -> else
SWITCH -> switch
CASE -> case
VAR_NAME -> var_name
BREAK -> break
DEFAULT -> default
COLON -> :
FOR -> for
WHILE -> while
CONTINUE -> continue
RETURN -> return
TRY -> try
CATCH -> catch
FINALLY -> finally
THROW -> throw
COMMA -> ,
STRING -> string
EQUALS -> =
PLUS -> +
MIN -> -
MULTIPLY -> *
DIVIDE -> /
MOD -> %
RIGHT -> <
LEFT -> >
AND -> &
OR -> |
XOR -> ^
QUESTION -> ?
DELETE -> delete
CONST -> const
LET -> let
VAR -> var
TRUE -> true
FALSE -> false
NAN -> NaN
NULL -> null
